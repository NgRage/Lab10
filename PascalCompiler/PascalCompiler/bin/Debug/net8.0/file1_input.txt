program Test;

var
  user : record
    id : integer;
    weight : real;
  end;
  
  x : integer;
  s : string;
  x : real;        // Повторное объявление переменной x 

begin
  (* Тут должно быть все норм *)
  x := 10 + 20 * (5 - 2);
  user.id := 1;
  
  with user do
  begin
    weight := 75.5;
    id := 5;
  end;

  (* Ошибки на лексику *)
  user.id := $100;                 // Запрещенный символ '$' 
  s := 'Строка не закрылась;       // Незакрытый строковый литерал 
  x := 999999999999999999999999;   // Число превышает лимит Integer 

  { Синтаксис и нейтрализация }
  x := + 5;           // + после :=
  with user x := 10;  // Пропущено ключевое слово 'do' 
  x := (10 + 5 * 2;   // Пропущена закрывающая круглая скобка 

  { Тут семантические ошибки }
  y := 50;            // Идентификатор 'y' не объявлен в блоке var 
  x := 'Hello';       // Несовпадение типов. Нельзя присвоить String в Integer 
  with x do           // Оператор 'with' ожидает тип record, а x - это Integer 
  begin
    id := 2;          // Неизвестый оператор, так как with do не поймал record
  end;

  user.id := 10;      // Парсер не ломается из-за ошибок и доходит до конца
end.
